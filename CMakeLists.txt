cmake_minimum_required(VERSION 2.8)
project(MoonServer)
add_subdirectory(lib)
add_subdirectory(src)
add_subdirectory(bin)
add_subdirectory(test)
add_subdirectory(include)
INCLUDE_DIRECTORIES("/usr/include/" "${CMAKE_BINARY_DIR}/include")
set(CMAKE_LIBRARY_PATH "${CMAKE_LIBRARY_PATH}" "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_CXX_FLAGS "-std=c++11 -Wall")

add_library(moon_server STATIC  src/command.cpp src/data.cpp src/motor.cpp src/zmqdata.cpp src/zmqserver.cpp src/zmqclient.cpp src/moonserver.cpp src/BoostComPort.cpp)
add_executable(MoonServer.exe src/main.cpp  src/command.cpp src/data.cpp src/motor.cpp src/zmqdata.cpp src/zmqserver.cpp src/zmqclient.cpp src/moonserver.cpp src/BoostComPort.cpp)
target_link_libraries(MoonServer.exe zmq pthread zmqpp-static boost_system)

enable_testing()
add_executable(test.exe test/test-data.cpp test/test-command.cpp test/test-zmqdata.cpp test/test-zmq.cpp)
target_link_libraries(test.exe gtest gtest_main pthread)
target_link_libraries(test.exe moon_server zmqpp-static zmq boost_system)
add_test(NAME "UnitTest" COMMAND "./bin/test.exe")

add_executable(TheSimple.exe src/SimpleMain.cpp  src/command.cpp src/data.cpp src/motor.cpp src/zmqdata.cpp src/zmqserver.cpp src/zmqclient.cpp src/moonserver.cpp src/BoostComPort.cpp) 
target_link_libraries(TheSimple.exe zmq pthread zmqpp-static curses c m boost_system)
