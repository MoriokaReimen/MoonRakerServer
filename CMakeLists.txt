cmake_minimum_required(VERSION 2.8)
project(MotorTest)
add_subdirectory(lib)
add_subdirectory(src)
add_subdirectory(bin)
add_subdirectory(test)
add_subdirectory(include)
INCLUDE_DIRECTORIES("/usr/include/" "${CMAKE_BINARY_DIR}/include")
set(CMAKE_LIBRARY_PATH "${CMAKE_LIBRARY_PATH}" "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_CXX_FLAGS "-std=c++11 -Wall")

add_library(moon_server STATIC src/rs232.cpp src/command.cpp src/data.cpp src/motor.cpp src/zmqdata.cpp)
add_executable(MoonServer.exe src/main.cpp src/rs232.cpp src/command.cpp src/data.cpp src/motor.cpp src/zmqdata.cpp)
target_link_libraries(MoonServer.exe zmq pthread)

enable_testing()
add_executable(test.exe test/test.cpp)
target_link_libraries(test.exe gtest gtest_main pthread)
target_link_libraries(test.exe moon_server)
add_test(NAME "UnitTest" COMMAND "./bin/test.exe")
